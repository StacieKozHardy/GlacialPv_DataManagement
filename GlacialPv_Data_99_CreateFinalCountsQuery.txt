select survey_id, survey_rep, start_dt_aklocal, num_seals, std_error
from surv_pv_gla.tbl_event e
left join surv_pv_gla.tbl_flyovers f
on e.id = f.event_id
left join surv_pv_gla.tbl_counts c
on e.id = c.event_id
where data_status_lku = 'C'
and survey_method_lku = 'T';

select polyid, track_rep, survey_dt, non_pup + pup as num_seals, 0.0 as std_error
from surv_pv_cst.summ_count_by_polyid
where polyid in (select polyid from surv_pv_gla.geo_glaciers);

select survey_id
from surv_pv_gla.tbl_event 
where id not in (select event_id from surv_pv_gla.tbl_counts);
